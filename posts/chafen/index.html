е·®еҲҶ

еҸ‘иЎЁдәҺ 2023-05-02 жӣҙж–°дәҺ 2024-05-11 еҲҶзұ»дәҺ CourseNotes

д»Җд№ҲжҳҜе·®еҲҶпјҢе·®еҲҶдёҺеүҚзјҖе’Ңзҡ„е…ізі»

е·®еҲҶд№ҹжҳҜдёҖз§ҚдјҳеҢ–з®—жі•гҖӮеҗҢж—¶пјҢе·®еҲҶжҳҜеүҚзјҖе’Ңзҡ„йҖҶиҝҗз®—пјҢд№ҹе°ұжҳҜиҜҙпјҢеүҚзјҖе’Ңд№ҹжҳҜе·®еҲҶзҡ„йҖҶиҝҗз®—гҖӮеӣ жӯӨпјҢз”ұеүҚзјҖе’Ңж•°з»„еҸҜд»ҘжұӮеҮәе·®еҲҶж•°з»„пјҢз”ұе·®еҲҶж•°з»„д№ҹеҸҜд»ҘжұӮеҮәеүҚзјҖе’Ңж•°з»„гҖӮ

еҒҮи®ҫж•°з»„ $a$ жҳҜеҺҹж•°з»„пјҢж•°з»„ $b$ жҳҜе·®еҲҶж•°з»„пјҢеҲҷпјҡ$b_i = a_i - a_{i-1}$

е·®еҲҶеҸҜд»Ҙз”ЁдәҺдҝ®ж”№ж•°з»„зҡ„ж“ҚдҪңгҖӮеӣ дёәеҒҮи®ҫжҲ‘йңҖиҰҒе°Ҷ $a_2$ $a_3$ $a_4$ йғҪеҠ дёҠ $2$ пјҢжӯӨж—¶иӢҘдҪҝз”ЁеүҚзјҖе’ҢеҲҷйңҖиҰҒеҶҚдҪҝз”ЁйҖ’жҺЁйҮҚж–°жұӮдёҖйҒҚеүҚзјҖе’ҢпјҢйқһеёёиҖ—ж—¶гҖӮиҝҷж—¶дҫҝеҸҜд»ҘдҪҝз”Ёе·®еҲҶж•°з»„гҖӮеҸҜд»ҘеҸ‘зҺ°пјҡ

\underbrace{b_1,\hspace{2mm} b_2}_{b_2 + 2 = b_1} ,\hspace{2mm} b_3,\hspace{2mm} \overbrace{b_4,\hspace{2mm} b_5}^{b_5 - 2 =b_4}

еӣ жӯӨпјҢеҸӘйңҖдҝ®ж”№е·®еҲҶж•°з»„дёӯзҡ„дёӨдёӘйЎ№пјҢз„¶еҗҺеҶҚйҖҡиҝҮеүҚзјҖе’ҢжҳҜе·®еҲҶзҡ„йҖҶиҝҗз®—пјҢеҚіеҸҜжұӮеҮәеҺҹж•°з»„пјҢд»ҺиҖҢе®ҢжҲҗж•°з»„зҡ„дҝ®ж”№гҖӮ

е…·дҪ“дҫӢйўҳпјҲжЁЎжқҝйўҳпјү

е·®еҲҶжЁЎжқҝйўҳ

йўҳзӣ®жҸҸиҝ°

з»ҷеҮәдёҖдёӘж•°еӯ—$n$иЎЁзӨәжңүдёӘж•°еӯ—пјҢ

з»ҷеҮә$n$ $n <= 10^5$дёӘж•ҙж•°$a_1$,$a_2$,вҖҰ$a_n$;

з»ҷеҮәдёҖдёӘж•°еӯ—$m$ $m <= 10^5$ жңү$m$дёӘдҝ®ж”№пјҡ
жҜҸж¬ЎиҜўй—®з»ҷеҮәдёүдёӘж•ҙж•°$s$,$e$,$h$пјҢдҪҝеҫ— $a_s,a_{s+1}вҖҰ.a_{e}$жҜҸдёҖдёӘж•°еҠ дёҠh

жңҖеҗҺз»ҷеҮәдёӨдёӘж•°еӯ— $start$,$end$гҖӮжұӮеҮә${ \sum_{i = start}^{end}} a_i $

иҫ“е…Ҙж јејҸ

з¬¬дёҖиЎҢдёҖдёӘж•ҙж•° $n$ иЎЁзӨәжңү$n$ дёӘж•°

з¬¬дәҢиЎҢ$n$дёӘж•ҙж•°$a_1$,$a_2$,вҖҰ$a_n$;

з¬¬дёүиЎҢдёҖдёӘж•ҙж•°$m$пјҢиЎЁзӨәжңү$m$дёӘдҝ®ж”№

жҺҘдёӢжқҘ$m$иЎҢжҜҸж¬ЎиҜўй—®з»ҷеҮәдёүдёӘж•ҙж•°$s$,$e$,$h$пјҢдҪҝеҫ— $a_s,a_{s+1}вҖҰ.a_{e}$жҜҸдёҖдёӘж•°еҠ дёҠh

жңҖеҗҺз»ҷеҮәдёӨдёӘж•°еӯ— $start$,$end$гҖӮжұӮеҮә${ \sum_{i = start}^{end}} a_i $

з¬¬дёҖж¬Ўдҝ®ж”№еәҸеҲ—еҸҳжҲҗ 2 3 3 4 5

з¬¬дәҢж¬Ўдҝ®ж”№еәҸеҲ—еҸҳжҲҗ 3 4 4 4 5

з¬¬дёүж¬Ўдҝ®ж”№еәҸеҲ—еҸҳжҲҗ 3 4 4 5 6

$0 <=$ $a_i$ е’Ң $h <= 10^4$

еҗҢдёҠпјҢ$b_{s-1} + h = b_s, b_{e + 1} - h = b_e$пјҢеҚіеҸҜдҪҝз”ЁеүҚзјҖе’ҢеҖ’йҖҖеӣһеҺҹж•°з»„.

#include <cstdio>
using namespace std ;

long long a[100002], b[100002], n, m, s, e, h, start, end, sum = 0 ;
int main()
{
    scanf("%lld", &n) ;
    for(int i = 1; i <= n; i ++)
    {
        scanf("%lld", &a[i]) ;
    }
    
    for(int i = 1; i <= n; i ++)
    {
        b[i] = a[i] - a[i - 1] ;
    }
    
    scanf("%lld", &m) ;
    for(int i = 1; i <= m; i ++)
    {
        scanf("%lld %lld %lld", &s, &e, &h) ;
        b[s] += h ;
        b[e + 1] -= h ;
    }
    for(int i = 1; i <= n; i ++)
    {
        a[i] = a[i - 1] + b[i] ;
    }
    
    scanf("%lld %lld", &start, &end) ;
    for(int i = start; i <= end; i ++)
    {
        sum += a[i] ;
    }
    printf("%lld\n", sum) ;
    
    return 0 ;
}
